// file: NotificationService.java
@Service
public class NotificationService {
    @Autowired private UserServiceClient userClient;
    @Autowired private EmailSender emailSender;

    public void sendOrderConfirmation(OrderCompletedEvent event) {
        // Fetch user contact info from User Service
        UserContact contact = userClient.getUserContact(event.getUserId());

        // Build email
        String subject = "Order " + event.getOrderId() + " Confirmed";
        String body = templateEngine.render("order-confirmation", event);

        // Send - this is where the NPE occurs (contact.getEmail() returns null)
        emailSender.sendEmail(contact.getEmail(), subject, body);
    }
}

--- Migration Script (the actual root cause) ---

-- file: V42__merge_legacy_profiles.sql
-- This migration merges the legacy 'user_profiles' table into the main 'users' table.
-- Legacy table used 'contact_email' column name; new table uses 'email'.

INSERT INTO users (id, name, phone, address, email, created_at)
SELECT
    lp.user_id,
    lp.full_name,
    lp.phone_number,
    lp.mailing_address,
    lp.email,            -- BUG: should be lp.contact_email (legacy column name)
    lp.registration_date
FROM legacy_profiles lp
WHERE lp.user_id NOT IN (SELECT id FROM users);

-- The legacy_profiles table has 'contact_email' NOT 'email'.
-- Using lp.email resolves to NULL because that column doesn't exist in legacy_profiles
-- (PostgreSQL doesn't error - it matches the alias from the INSERT column list).
-- Result: 342 migrated users have NULL email.

The Notification Service code is working correctly - it should fail loudly when
