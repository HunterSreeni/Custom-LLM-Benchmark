// file: src/services/orderService.js

async function createOrder(userId, items) {
    const validatedItems = await validateItems(items);

    // Enrich order with user data from multiple sources
    const enrichedData = await enrichOrder(userId);

    const order = {
        userId,
        items: validatedItems,
        ...enrichedData,
        status: 'pending',
        createdAt: new Date(),
    };

    return await db.orders.insert(order);
}

// file: src/services/enrichmentPipeline.js

async function enrichOrder(userId) {
    // BUG: Uses Promise.all but one of the inner functions has its own
    // fire-and-forget promise that isn't included in the Promise.all.

    const [profile, loyalty, shipping] = await Promise.all([
        fetchUserProfile(userId),
        fetchLoyaltyPoints(userId),
        fetchShippingPrefs(userId),
    ]);

    return { profile, loyalty, shipping };
}

async function fetchUserProfile(userId) {
    const response = await httpClient.get(`user-api.internal/users/${userId}`);
    return response.data;
}

async function fetchLoyaltyPoints(userId) {
    const response = await httpClient.get(`loyalty-api.internal/points/${userId}`);

    // BUG: This fire-and-forget promise is NOT awaited and NOT caught.
    // If updateLastAccessed fails, it creates an unhandled rejection.
    // However, the VISIBLE error "Failed to enrich user data" actually
    // comes from the HTTP 503 above - Promise.all rejects fast.
    //
    // The REAL unhandled rejection is from updateLastAccessed, which runs
    // AFTER Promise.all has already rejected. Since the parent try/catch
    // in createOrder only catches the Promise.all rejection, this orphaned
    // promise has no handler.
    updateLastAccessed(userId);  // fire-and-forget - no await, no .catch()

    return response.data;
}

async function updateLastAccessed(userId) {
    // This runs independently. If it fails, nothing catches it.
    const result = await httpClient.put(`loyalty-api.internal/last-access/${userId}`, {
        timestamp: new Date().toISOString(),
    });

    if (!result.ok) {
        throw new Error("Failed to enrich user data");  // Misleading error message
    }
    return result;
}

async function fetchShippingPrefs(userId) {
    const response = await httpClient.get(`shipping-api.internal/prefs/${userId}`);
    return response.data;
}

// file: src/routes/orders.js

router.post('/api/orders', async (req, res) => {
    try {
        const order = await createOrder(req.body.userId, req.body.items);
        res.status(201).json(order);
    } catch (err) {
        // This catches the Promise.all rejection (HTTP 503 from loyalty service)
        // but NOT the orphaned updateLastAccessed promise
        logger.error(`Order creation failed: ${err.message}`);
        res.status(500).json({ error: 'Order creation failed' });
    }
});
